module mod_13 (
    input wire clk,
    input wire rst_n,
    input wire shift_in,
    input wire t,
    output reg [7:0] shift_reg,
    output reg q
);
    always @(posedge clk) begin
        if (!rst_n)
            shift_reg <= 8'b0;
        else
            shift_reg <= {shift_reg[6:0], shift_in};
    end
    always @(posedge clk) begin
        if (!rst_n)
            q <= 1'b0;
        else if (t)
            q <= ~q;
    end
endmodule
